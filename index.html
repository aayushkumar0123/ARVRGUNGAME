    <script>
        // --- GAME STATE & VARIABLES ---
        let gameState = 'loading'; // loading, start, playing, gameover, ended
        let score = 0;
        let misses = 0;
        const MAX_MISSES = 5;

        // Audio Context for Procedural Sounds
        let audioCtx;

        // DOM Elements
        const uiLoading = document.getElementById('loading-screen');
        const uiStart = document.getElementById('start-screen');
        const uiHUD = document.getElementById('hud');
        const uiGameOver = document.getElementById('game-over-screen');
        const uiSessionEnded = document.getElementById('session-ended-screen');
        const scoreDisplay = document.getElementById('score-display');
        const missDisplay = document.getElementById('miss-display');
        const finalScoreDisplay = document.getElementById('final-score');
        const videoElement = document.getElementById('webcam-video');
        const canvasElement = document.getElementById('game-canvas');

        // --- THREE.JS SETUP ---
        const scene = new THREE.Scene();
        // Make the background purely black so it hides the real camera feed behind it
        scene.background = new THREE.Color(0x000000);
        
        // Camera setup - looking down the negative Z axis
        const camera = new THREE.PerspectiveCamera(60, window.innerWidth / window.innerHeight, 0.1, 1000);
        camera.position.set(0, 2, 5);

        // Remove alpha: true so the canvas acts as a solid black wall over the video
        const renderer = new THREE.WebGLRenderer({ canvas: canvasElement, antialias: true });
        renderer.setSize(window.innerWidth, window.innerHeight);
        renderer.setPixelRatio(window.devicePixelRatio);

        // Lighting
        const ambientLight = new THREE.AmbientLight(0xffffff, 0.5);
        scene.add(ambientLight);
        const dirLight = new THREE.DirectionalLight(0xffffff, 0.8);
        dirLight.position.set(0, 10, 5);
        scene.add(dirLight);

        // Endless Path (Moving Grid)
        const gridHelper = new THREE.GridHelper(40, 40, 0x00ffff, 0x004444);
        gridHelper.position.y = -1;
        scene.add(gridHelper);

        // VR Environment Stars/Space Dust
        const starsGeo = new THREE.BufferGeometry();
        const starsCount = 500;
        const starPosArray = new Float32Array(starsCount * 3);
        for(let i = 0; i < starsCount; i++) {
            starPosArray[i*3] = (Math.random() - 0.5) * 100;     // X
            starPosArray[i*3+1] = (Math.random() - 0.5) * 100;   // Y
            starPosArray[i*3+2] = (Math.random() - 1.0) * 100;   // Z
        }
        starsGeo.setAttribute('position', new THREE.BufferAttribute(starPosArray, 3));
        const starsMat = new THREE.PointsMaterial({size: 0.15, color: 0x00ffff, transparent: true, opacity: 0.5});
        const starMesh = new THREE.Points(starsGeo, starsMat);
        scene.add(starMesh);

        // Game Objects Collections
        const targets = [];
        const bullets = [];
        const particles = [];

        // Materials
        const leftGunMat = new THREE.MeshStandardMaterial({ color: 0x00ffff, emissive: 0x00aaaa });
        const rightGunMat = new THREE.MeshStandardMaterial({ color: 0xff00ff, emissive: 0xaa00aa });
        const bulletMat = new THREE.MeshBasicMaterial({ color: 0xffffff });
        const targetMatLeft = new THREE.MeshStandardMaterial({ color: 0x00ffff, emissive: 0x005555, wireframe: true });
        const targetMatRight = new THREE.MeshStandardMaterial({ color: 0xff00ff, emissive: 0x550055, wireframe: true });

        // Gun Models (Simple boxes)
        function createGun(material) {
            const group = new THREE.Group();
            const barrel = new THREE.Mesh(new THREE.BoxGeometry(0.2, 0.2, 1), material);
            barrel.position.z = -0.5;
            const handle = new THREE.Mesh(new THREE.BoxGeometry(0.2, 0.5, 0.2), material);
            handle.position.y = -0.25;
            group.add(barrel);
            group.add(handle);
            // Hide initially until hand is detected
            group.visible = false;
            scene.add(group);
            return group;
        }

        const gunLeft = createGun(leftGunMat);
        const gunRight = createGun(rightGunMat);

        // --- AUDIO GENERATION ---
        function initAudio() {
            if (!audioCtx) {
                audioCtx = new (window.AudioContext || window.webkitAudioContext)();
            }
            if (audioCtx.state === 'suspended') {
                audioCtx.resume();
            }
        }

        function playShootSound() {
            if (!audioCtx) return;
            const osc = audioCtx.createOscillator();
            const gain = audioCtx.createGain();
            osc.connect(gain);
            gain.connect(audioCtx.destination);
            osc.type = 'square';
            osc.frequency.setValueAtTime(880, audioCtx.currentTime);
            osc.frequency.exponentialRampToValueAtTime(110, audioCtx.currentTime + 0.1);
            gain.gain.setValueAtTime(0.05, audioCtx.currentTime);
            gain.gain.exponentialRampToValueAtTime(0.001, audioCtx.currentTime + 0.1);
            osc.start();
            osc.stop(audioCtx.currentTime + 0.1);
        }

        function playExplosionSound() {
            if (!audioCtx) return;
            const bufferSize = audioCtx.sampleRate * 0.2; // 0.2 seconds
            const buffer = audioCtx.createBuffer(1, bufferSize, audioCtx.sampleRate);
            const data = buffer.getChannelData(0);
            for (let i = 0; i < bufferSize; i++) {
                data[i] = Math.random() * 2 - 1;
            }
            const noise = audioCtx.createBufferSource();
            noise.buffer = buffer;
            const gain = audioCtx.createGain();
            // Filter to make it sound deeper
            const filter = audioCtx.createBiquadFilter();
            filter.type = 'lowpass';
            filter.frequency.value = 1000;
            
            noise.connect(filter);
            filter.connect(gain);
            gain.connect(audioCtx.destination);
            
            gain.gain.setValueAtTime(0.1, audioCtx.currentTime);
            gain.gain.exponentialRampToValueAtTime(0.001, audioCtx.currentTime + 0.2);
            noise.start();
        }

        function playErrorSound() {
            if (!audioCtx) return;
            const osc = audioCtx.createOscillator();
            const gain = audioCtx.createGain();
            osc.connect(gain);
            gain.connect(audioCtx.destination);
            osc.type = 'sawtooth';
            osc.frequency.setValueAtTime(150, audioCtx.currentTime);
            gain.gain.setValueAtTime(0.1, audioCtx.currentTime);
            gain.gain.linearRampToValueAtTime(0, audioCtx.currentTime + 0.3);
            osc.start();
            osc.stop(audioCtx.currentTime + 0.3);
        }

        // --- GAMEPLAY MECHANICS ---

        function spawnTarget() {
            if (gameState !== 'playing') return;
            
            const isLeft = Math.random() > 0.5;
            const geo = new THREE.OctahedronGeometry(0.8, 0);
            const mat = isLeft ? targetMatLeft : targetMatRight;
            const target = new THREE.Mesh(geo, mat);
            
            // Spawn far away on Z, placed in left or right lane
            target.position.set(isLeft ? -2.5 : 2.5, 1, -40);
            
            // Store lane info for logic. Slower base speed and variance.
            target.userData = { isLeft: isLeft, speed: 0.08 + (Math.random() * 0.05) };
            
            scene.add(target);
            targets.push(target);
        }

        function shoot(gunOrigin, isLeft) {
            playShootSound();
            const geo = new THREE.SphereGeometry(0.1, 8, 8);
            const bullet = new THREE.Mesh(geo, bulletMat);
            
            // Position bullet at the tip of the gun
            bullet.position.copy(gunOrigin.position);
            // Move slightly forward
            bullet.position.z -= 1.5;
            
            bullet.userData = { isLeft: isLeft, speed: 1.0 };
            scene.add(bullet);
            bullets.push(bullet);
        }

        function createExplosion(position, color) {
            playExplosionSound();
            const particleCount = 20;
            const geo = new THREE.BufferGeometry();
            const posArray = new Float32Array(particleCount * 3);
            const velArray = [];

            for(let i=0; i < particleCount; i++) {
                posArray[i*3] = position.x;
                posArray[i*3+1] = position.y;
                posArray[i*3+2] = position.z;
                velArray.push({
                    x: (Math.random() - 0.5) * 0.5,
                    y: (Math.random() - 0.5) * 0.5,
                    z: (Math.random() - 0.5) * 0.5
                });
            }
            geo.setAttribute('position', new THREE.BufferAttribute(posArray, 3));
            
            const mat = new THREE.PointsMaterial({ size: 0.2, color: color, transparent: true });
            const pSystem = new THREE.Points(geo, mat);
            
            scene.add(pSystem);
            particles.push({ system: pSystem, velocities: velArray, life: 1.0 });
        }

        function updateGame() {
            if (gameState !== 'playing') return;

            // Move Grid to simulate speed (slower)
            gridHelper.position.z += 0.08;
            if (gridHelper.position.z > 2) gridHelper.position.z = 0;

            // Animate Starfield (slower)
            const starPositions = starMesh.geometry.attributes.position.array;
            for(let i = 2; i < starsCount * 3; i += 3) {
                starPositions[i] += 0.15; // Move stars towards camera
                if(starPositions[i] > 10) {
                    starPositions[i] = -80; // Reset far back
                }
            }
            starMesh.geometry.attributes.position.needsUpdate = true;

            // Random Target Spawning (slower spawn rate to match slower speed)
            if (Math.random() < 0.01 + (score * 0.0003)) {
                spawnTarget();
            }

            // Update Bullets
            for (let i = bullets.length - 1; i >= 0; i--) {
                const b = bullets[i];
                b.position.z -= b.userData.speed;
                // Remove if too far
                if (b.position.z < -50) {
                    scene.remove(b);
                    bullets.splice(i, 1);
                }
            }

            // Update Targets & Check Collisions
            for (let i = targets.length - 1; i >= 0; i--) {
                const t = targets[i];
                t.position.z += t.userData.speed;
                t.rotation.y += 0.05;
                t.rotation.x += 0.02;

                let hit = false;
                
                // Collision with bullets
                for (let j = bullets.length - 1; j >= 0; j--) {
                    const b = bullets[j];
                    const dist = t.position.distanceTo(b.position);
                    
                    // Simple bounding sphere collision
                    if (dist < 1.0) {
                        // Explosion color matches target
                        createExplosion(t.position, t.userData.isLeft ? 0x00ffff : 0xff00ff);
                        
                        // Remove bullet
                        scene.remove(b);
                        bullets.splice(j, 1);
                        
                        // Remove target
                        scene.remove(t);
                        targets.splice(i, 1);
                        
                        score += 10;
                        scoreDisplay.innerText = score;
                        hit = true;
                        break;
                    }
                }

                if (!hit) {
                    // Missed target condition: passes behind camera
                    if (t.position.z > camera.position.z) {
                        scene.remove(t);
                        targets.splice(i, 1);
                        misses++;
                        missDisplay.innerText = misses;
                        playErrorSound();
                        
                        if (misses >= MAX_MISSES) {
                            triggerGameOver();
                        }
                    }
                }
            }

            // Update Particles
            for (let i = particles.length - 1; i >= 0; i--) {
                const p = particles[i];
                p.life -= 0.02;
                
                if (p.life <= 0) {
                    scene.remove(p.system);
                    p.system.geometry.dispose();
                    p.system.material.dispose();
                    particles.splice(i, 1);
                    continue;
                }

                const positions = p.system.geometry.attributes.position.array;
                for (let j = 0; j < p.velocities.length; j++) {
                    positions[j*3] += p.velocities[j].x;
                    positions[j*3+1] += p.velocities[j].y;
                    positions[j*3+2] += p.velocities[j].z;
                }
                p.system.geometry.attributes.position.needsUpdate = true;
                p.system.material.opacity = p.life;
            }
        }

        function triggerGameOver() {
            gameState = 'gameover';
            uiHUD.classList.add('hidden');
            uiGameOver.classList.remove('hidden');
            finalScoreDisplay.innerText = score;
            
            // Hide guns
            gunLeft.visible = false;
            gunRight.visible = false;
        }

        function resetGame() {
            // Clean up scene
            targets.forEach(t => scene.remove(t));
            bullets.forEach(b => scene.remove(b));
            particles.forEach(p => { scene.remove(p.system); p.system.geometry.dispose(); p.system.material.dispose(); });
            
            targets.length = 0;
            bullets.length = 0;
            particles.length = 0;

            score = 0;
            misses = 0;
            scoreDisplay.innerText = score;
            missDisplay.innerText = misses;
            
            gameState = 'playing';
            uiGameOver.classList.add('hidden');
            uiHUD.classList.remove('hidden');
        }

        // --- HAND TRACKING (MEDIAPIPE) ---
        
        // Track hand states to prevent continuous firing
        const handStates = {
            left: { isPinching: false, lastActive: 0 },
            right: { isPinching: false, lastActive: 0 }
        };

        function onResults(results) {
            if (gameState !== 'playing') return;

            // Reset gun visibility. Will turn true if detected.
            gunLeft.visible = false;
            gunRight.visible = false;

            // Relaxed condition to ensure tracking doesn't drop if handedness label is briefly lost
            if (results.multiHandLandmarks && results.multiHandLandmarks.length > 0) {
                
                // 1. Gather and sort hands by their visual screen position (left to right)
                let detectedHands = [];
                for (let i = 0; i < results.multiHandLandmarks.length; i++) {
                    detectedHands.push({
                        landmarks: results.multiHandLandmarks[i],
                        visualX: 1.0 - results.multiHandLandmarks[i][0].x // Mirror X
                    });
                }
                detectedHands.sort((a, b) => a.visualX - b.visualX);

                // 2. Process each detected hand
                for (let i = 0; i < detectedHands.length; i++) {
                    const landmarks = detectedHands[i].landmarks;
                    
                    // Use palm center (base of middle finger) instead of wrist for a better anchor
                    const palm = landmarks[9]; 
                    const visualX = detectedHands[i].visualX;
                    
                    // Apply a vertical offset so you don't have to hold your hands up so high.
                    // Subtracting from Y pushes the virtual gun higher on the screen relative to your physical hand.
                    const yOffset = 0.2; 
                    const visualY = palm.y - yOffset;
                    
                    // Assign guns reliably: Leftmost hand gets Left Gun, rightmost gets Right Gun.
                    // If only one hand is visible, assign based on which side of the screen it's on.
                    let isLeftGun;
                    if (detectedHands.length === 2) {
                        isLeftGun = (i === 0);
                    } else {
                        isLeftGun = visualX < 0.5;
                    }

                    const gun = isLeftGun ? gunLeft : gunRight;
                    const state = isLeftGun ? handStates.left : handStates.right;
                    
                    // 3. Perfect mathematical mapping to 3D Space
                    const aspect = window.innerWidth / window.innerHeight;
                    const distance = camera.position.z - 2; // Distance from camera (5) to gun Z (2) = 3
                    const vFov = camera.fov * Math.PI / 180;
                    const planeHeight = 2 * Math.tan(vFov / 2) * distance;
                    const planeWidth = planeHeight * aspect;
                    
                    // Position guns exactly where your hands are on screen
                    gun.position.set(
                        (visualX - 0.5) * planeWidth,
                        -(visualY - 0.5) * planeHeight,
                        2
                    );
                    
                    // Add slight rotation based on movement direction
                    const ndcX = (visualX * 2) - 1;
                    const ndcY = -(visualY * 2) + 1;
                    gun.rotation.y = -ndcX * 0.3;
                    gun.rotation.x = ndcY * 0.3;
                    
                    gun.visible = true;

                    // 4. Improved Gesture Recognition: Pinch (using 2D distance for stability)
                    const thumb = landmarks[4];
                    const index = landmarks[8];
                    const dist = Math.hypot(thumb.x - index.x, thumb.y - index.y);
                    
                    const pinchThreshold = 0.06; // Adjusted threshold for 2D distance

                    if (dist < pinchThreshold) {
                        if (!state.isPinching) {
                            shoot(gun, isLeftGun);
                            state.isPinching = true;
                        }
                    } else {
                        state.isPinching = false;
                    }
                }
            }
        }

        const hands = new Hands({locateFile: (file) => {
            return `https://cdn.jsdelivr.net/npm/@mediapipe/hands/${file}`;
        }});
        
        hands.setOptions({
            maxNumHands: 2,
            modelComplexity: 1,
            minDetectionConfidence: 0.6,
            minTrackingConfidence: 0.6
        });
        
        hands.onResults(onResults);

        // --- CAMERA SETUP & RENDER LOOP ---

        const cameraUtils = new Camera(videoElement, {
            onFrame: async () => {
                await hands.send({image: videoElement});
            },
            width: 1280,
            height: 720
        });

        cameraUtils.start().then(() => {
            // Camera started, models loaded
            uiLoading.classList.add('hidden');
            uiStart.classList.remove('hidden');
            gameState = 'start';
        }).catch(err => {
            console.error("Camera error:", err);
            uiLoading.innerHTML = `<h1 class="text-2xl text-red-500">Camera Access Denied or Unavailable.</h1>`;
        });

        // Animation Loop
        function animate() {
            requestAnimationFrame(animate);
            updateGame();
            
            // Only render 3D scene if not ended
            if (gameState !== 'ended') {
                renderer.render(scene, camera);
            }
        }
        animate();

        // Window Resize Handler
        window.addEventListener('resize', () => {
            camera.aspect = window.innerWidth / window.innerHeight;
            camera.updateProjectionMatrix();
            renderer.setSize(window.innerWidth, window.innerHeight);
        });

        // --- UI BUTTON LISTENERS ---

        document.getElementById('btn-start').addEventListener('click', () => {
            initAudio();
            uiStart.classList.add('hidden');
            uiHUD.classList.remove('hidden');
            gameState = 'playing';
        });

        document.getElementById('btn-retry').addEventListener('click', () => {
            resetGame();
        });

        document.getElementById('btn-end').addEventListener('click', () => {
            gameState = 'ended';
            uiGameOver.classList.add('hidden');
            uiSessionEnded.classList.remove('hidden');
            
            // Clear canvas rendering
            renderer.clear();
            canvasElement.style.display = 'none';
        });

        document.getElementById('btn-restart-app').addEventListener('click', () => {
            // Bring back to start screen
            uiSessionEnded.classList.add('hidden');
            canvasElement.style.display = 'block';
            resetGame();
            gameState = 'start';
            uiStart.classList.remove('hidden');
            uiHUD.classList.add('hidden');
        });

    </script>
